// FullCommand class
template <typename Operand>
class FullCommand {
public:
    void addArgument(const std::string& key, Operand value);
    // Other methods for manipulating the command.
};

// Parser class

public:
    FullCommand<Operand> parse(const std::string& input);
};

public:
    void registerCommand(const std::string& commandName, const FullCommand<Operand>& command);

};
